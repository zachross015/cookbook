

\begin{recipe}[
        time={30 minutes},
        servings=6,
        source={https://tasty.co/recipe/cinnamon-roll-french-toast-bake},
        calories=587,
        fat=21,
        carbs=87,
        sugar=52,
        protein=14
    ]{Cinnamon Roll French Toast}


    \ingredients{%
        2 & tubes & cinnamon roll, refigerated, with icing \\
        4 & tbsp & butter, melted \\
        4 & & eggs\\
        1/2 & cup & milk \\
        2 & tsp & cinnamon \\
        2 & tsp & vanilla \\
        1/2 & cup & maple syrup
    }

    \preparation{%
        \item Cut each cinnamon roll into 8 pieces.
        \item In a separate bowl, whisk the eggs, milk, vanilla, and cinnamon.
            Set it aside.
        \item Pour the butter on the bottom of a $9 \times 13$ inch ($22\text{cm} \times
            33\text{cm}$) glass baking dish. Spread the cinnamon roll pieces evenly over
            the butter. The pour the mixture over the cinnamon rolls. Pour 1/2 cup of
            maple syrup over the mixture.
        \item Bake at 375\degree F (190\degree C) for 25 min.
        \item Top with the icing from the cinnamon rolls.
    }

\end{recipe}
